- image resize
- invoice generation
- email sending
- video processing

for msg := range msgs {
	if err := doHeavyJob(msg.Body); err != nil {
		_ = msg.Nack(false, true)
		continue
	}

	time.Sleep(2 * time.Second)
	_ = msg.Ack(false)
}
